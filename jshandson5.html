var Game_Winner = (S) =>
{
let countA=0;
let countD=0;
for(let i=0;i < S.length;i++) {
if(S[i]==='A') {
countA+=1;
}
else {
countD+=1;
}
}
if(countA>countD) {
return "Aditya";
}
else if(countA< countD) {
return "Danish";
}
else {
return "Draw";

{

for (let i=N-1; i>=0; i--)
{
revstr += str[i];
}
if (str === revstr)

return "True";
}
else{
return "False";
}
}
